# 采用多表关联
# 图片表单只存储 图片名字 图片地址 以及图片编号
# 新闻和酒店各一个 标题 内容 编号
# 中间表达记录 图片编号 新闻或酒店编号  对应的图片描述

# 简短设置同样需要通过json来设置

最后获取到的数据都将通过中间表单来获取

# 账号数据库
id samllint
username varchar(20) unique key
power enum('admin', 'root') default 'admin'
date

# 图片数据
id 编号
name 文件名
path 真实地址
url 引用地址
desc 可以空
status 状态 默认正常 当管理员删除时会在这里标记处理 但不做真实删除 1 表示未删除 0 表示删除
used 0 或 1  当前该图片资源是否被使用，1表示在使用 0 表示没有被使用
date 上传时间
dir  图片所属文件夹

# 关联表
id 编号
articleid 文章编号
imageid 图片编号

# 招聘表
id 编号 TINYINT(2) UNSIGNED NOT NULL PRIMARY KEY
name 职位 TINYTEXT NOT NULL
note  职位简要 TINYTEXT NOT NULL
loc  工作地址 TINYTEXT
user  管理员 TINYTEXT
req  要求 TEXT
des 描述 TEXT
date 时间 TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP

# 新闻表
id 编号 VARCHAR(32) NOT NULL PRIMARY KEY
title 标题 TINYTEXT NOT NULL
poster  封面id SMALLINT(5) NOT NULL
images 新闻画面 TINYTEXT NOT NULL
content 新闻内容 TEXT NOT NULL
username 发布人员 TINYTEXT
date 时间
